/* style.css */ * { margin: 0; padding: 0; box-sizing: border-box; } body { background-color: #1e1e1e; display: flex; justify-content: center; align-items: center; height: 100vh; } .game-container { position: relative; } canvas { background-color: #222; display: block; border: 2px solid white; } #score { position: absolute; top: 10px; left: 10px; color: white; font-size: 20px; font-family: Arial, sans-serif; } // script.js const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // Set up the canvas size canvas.width = 800; canvas.height = 600; // Player settings const player = { x: canvas.width / 2, y: canvas.height / 2, radius: 15, color: 'green', speed: 5, angle: 0 }; // Bullet settings const bullets = []; const bulletSpeed = 7; // Target settings const target = { x: Math.random() * canvas.width, y: Math.random() * canvas.height, radius: 20, color: 'red' }; // Score variable let score = 0; // Listen to mouse movement for aiming canvas.addEventListener('mousemove', (event) => { const dx = event.clientX - canvas.offsetLeft - player.x; const dy = event.clientY - canvas.offsetTop - player.y; player.angle = Math.atan2(dy, dx); }); // Listen for mouse click to shoot canvas.addEventListener('click', () => { const bullet = { x: player.x, y: player.y, radius: 5, color: 'white', angle: player.angle }; bullets.push(bullet); }); // Function to update the player's position function movePlayer() { // No movement logic yet, the player is fixed in the center } // Function to update the bullet's position function moveBullets() { for (let i = 0; i < bullets.length; i++) { const bullet = bullets[i]; bullet.x += Math.cos(bullet.angle) * bulletSpeed; bullet.y += Math.sin(bullet.angle) * bulletSpeed; // Remove bullet if it goes off-screen if (bullet.x < 0 || bullet.x > canvas.width || bullet.y < 0 || bullet.y > canvas.height) { bullets.splice(i, 1); i--; } } } // Function to check if a bullet hits the target function checkCollision() { for (let i = 0; i < bullets.length; i++) { const bullet = bullets[i]; const dx = bullet.x - target.x; const dy = bullet.y - target.y; const distance = Math.sqrt(dx * dx + dy * dy); if (distance < bullet.radius + target.radius) { // If collision happens, reset target position and update score target.x = Math.random() * canvas.width; target.y = Math.random() * canvas.height; score++; document.getElementById('score').textContent = `Score: ${score}`; // Remove the bullet after hit bullets.splice(i, 1); i--; } } } // Function to draw the player function drawPlayer() { ctx.beginPath(); ctx.arc(player.x, player.y, player.radius, 0, Math.PI * 2); ctx.fillStyle = player.color; ctx.fill(); ctx.closePath(); } // Function to draw the bullets function drawBullets() { for (let i = 0; i < bullets.length; i++) { const bullet = bullets[i]; ctx.beginPath(); ctx.arc(bullet.x, bullet.y, bullet.radius, 0, Math.PI * 2); ctx.fillStyle = bullet.color; ctx.fill(); ctx.closePath(); } } // Function to draw the target function drawTarget() { ctx.beginPath(); ctx.arc(target.x, target.y, target.radius, 0, Math.PI * 2); ctx.fillStyle = target.color; ctx.fill(); ctx.closePath(); } // Game loop to update the canvas function gameLoop() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Clear the canvas movePlayer(); moveBullets(); checkCollision(); drawPlayer(); drawBullets(); drawTarget(); requestAnimationFrame(gameLoop); // Recursively call the game loop } // Start the game loop gameLoop();
